// ProductCard.jsx
const [imageState, setImageState] = useState('loading');

  {imageState === 'loading' && <SkeletonLoader />}

## 2. Katalógus Filter Perzisztencia — Implementáció

### Frontend (2 nap)

**Fájlok:**
- `src/stores/app-store.jsx` (új metódusok)
- `src/views/catalog-world-view.jsx` (integrálás)

**App-store új logika:**
```javascript
// Versioned storage
const FILTER_VERSION = 1;
saveCatalogFilters(filters) {
  localStorage.setItem('spaceos_catalog_filters_v' + FILTER_VERSION, 
    JSON.stringify({filters, timestamp: Date.now()}));
}
loadCatalogFilters() {
  const stored = localStorage.getItem('spaceos_catalog_filters_v' + FILTER_VERSION);
  if (!stored) return null;
  const data = JSON.parse(stored);
  // 24 óránál régebbi cache-t dob
  if (Date.now() - data.timestamp > 86400000) return null;
  return data.filters;
}
```

**UX flow:**
1. Komponens mount → `useEffect` betölti filter-t
2. Filter change → 300ms debounce után auto-save
3. Nav badge: `{activeFilterCount > 0 && <Badge>{activeFilterCount}</Badge>}`
4. "Szűrők törlése" gomb → `clearCatalogFilters()` + UI reset

**Kockázatok:**
- LocalStorage kvóta betelt (5MB limit) → Try-catch wrapper + fallback sessionStorage
- Multi-tab szinkron probléma → `storage` event listener kell (2 óra extra)

**Backend:** ❌ Nincs szükség

## 3. Assembly Drag-and-Drop — Implementáció

### Backend (2 nap)

**Új endpoint:**
```
PATCH /api/v1/work-orders/{id}/assembly-sequence
Body: { operations: [{id: 'op-1', sequence: 1}, {id: 'op-2', sequence: 2}] }
Response: 200 OK + updated WorkOrder object
```

**Validációk:**
- Minden operation.id létezik az adott work order-ben
- Sequence számok folytonosak (1, 2, 3... gap nélkül)
- Work order státusza nem "completed"

**Adatmodell:**
```csharp
// WorkOrderOperation.cs
public class WorkOrderOperation {
  public string Id { get; set; }
  public int Sequence { get; set; } // ÚJ mező, migration kell
  public string Description { get; set; }
  public TimeSpan EstimatedDuration { get; set; }
}
```

### Frontend (3 nap)

**Library:** `@dnd-kit/core` + `@dnd-kit/sortable` (34KB gzipped)

**Komponens:**
```jsx
// assembly.jsx refactor
import {DndContext, closestCenter} from '@dnd-kit/core';
import {SortableContext, useSortable, verticalListSortingStrategy} from '@dnd-kit/sortable';

function OperationItem({operation}) {
  const {attributes, listeners, setNodeRef, transform, transition} = useSortable({id: operation.id});
  // Drag handle csak kis képernyőn long-press
  return (
    <div ref={setNodeRef} style={{transform, transition}}>
      <div {...listeners} {...attributes}>⋮⋮</div> {/* Drag handle */}
      {operation.description}
    </div>
  );
}
```

**State flow:**
1. Drag end → local state frissül (optimistic UI)
2. API call → `PATCH /assembly-sequence`
3. Hiba esetén → rollback + toast "Mentés sikertelen"

**Kockázatok:**
- Egyidejű szerkesztés (2 user ugyanazt a work order-t rendezi) → Backend timestamp-based conflict detection kell (1 nap extra)
- Touch device gesztusok ütköznek (scroll vs drag) → Explicit "Szerkesztés mód" toggle (1 nap extra)
