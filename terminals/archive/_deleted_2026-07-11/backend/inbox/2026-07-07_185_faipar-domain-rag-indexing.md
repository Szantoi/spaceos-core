2. **Embedding Function**: XenovaEmbeddingFunction (all-MiniLM-L6-v2, 384 dim)
3. **Chunking Strategy**: Markdown section-based (## header) vagy fixed 1000 token chunks

**Code Location**: `spaceos-nexus/knowledge-service/src/`

**Implementációs lépések**:

```typescript
// 1. Import dependencies
import { getOrCreateCollection } from './vectorStore';
import { XenovaEmbeddingFunction } from './xenovaEmbedding';
import fs from 'fs';

// 2. Read files
const files = [
  '/opt/spaceos/docs/faipari_gyartasszervezes_rag.md',
  '/opt/spaceos/docs/faipari_muszaki_dokumentacio_rag.md',
  '/opt/spaceos/docs/woodwork_domain.md'
];

// 3. Chunk documents (markdown section-based)
function chunkMarkdown(content: string, filePath: string) {
  // Split by ## headers
  const sections = content.split(/^## /m);
  return sections.map((section, idx) => ({
    id: `${path.basename(filePath)}-chunk-${idx}`,
    text: section.trim(),
    metadata: {
      source: filePath,
      category: 'faipar-domain',
      section: idx
    }
  }));
}

// 4. Upsert to ChromaDB
const embeddingFunction = new XenovaEmbeddingFunction();
const collection = await getOrCreateCollection('spaceos-knowledge', embeddingFunction);

for (const filePath of files) {
  const content = fs.readFileSync(filePath, 'utf8');
  const chunks = chunkMarkdown(content, filePath);

  await collection.add({
    ids: chunks.map(c => c.id),
    documents: chunks.map(c => c.text),
    metadatas: chunks.map(c => c.metadata)
  });

  console.log(`✅ Indexed: ${path.basename(filePath)} (${chunks.length} chunks)`);
}
```
