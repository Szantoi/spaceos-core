**Example usage:**
```tsx
import { useGetEacCalculation } from '@/api/kontrolling';

export function EACCalculationWidget({ projectId }: { projectId: string }) {
  const { data, isLoading, error } = useGetEacCalculation({ projectId });

  if (isLoading) return <Skeleton />;
  if (error) return <ErrorAlert error={error} />;

  return (
    <Card>
      <CardHeader>EAC Calculation</CardHeader>
      <CardContent>
        {data?.categories.map(cat => (
          <div key={cat.category}>
            <span>{cat.category}:</span>
            <span>Planned: {cat.planned}</span>
            <span>Actual: {cat.actual}</span>
            <span>EAC: {cat.eac}</span>
          </div>
        ))}
      </CardContent>
    </Card>
  );
}
```
